// 当前所在页面的 PDFPageRender 对象
textSelectionTool.pageRender

自定义创建 squiggly controller 示例

class CreateSquigglyOnSelectedTextController extends UIExtension.Controller {
    static getName() {
        return 'CreateSquigglyOnSelectedTextController';
    }
    handle() {
        var tooltipLayer = this.component.getClosestComponentByType('tooltip-layer');
        var textSelectionTool = tooltipLayer.getCurrentSelectionTool();
        textSelectionTool.getSelectionInfo().then((selectionInfo)=> {
            this.createTextMarkupAnnot(textSelectionTool.pageRender, selectionInfo, 'squiggly').then(() => {
                tooltipLayer.hide(); // 创建完成后就隐藏 tooltip
            });
        })
    }
    // 根据类型值创建不同的 textmarkup annotation: underline, strikeout, highlight, caret, squiggly
    createTextMarkupAnnot(pageRender, selectionInfo, type) {
        return pageRender.getPDFPage().then(page => {
            return page
                .addAnnot({
                    type: type,
                    rect: {
                        left: 0,
                        right: 0,
                        top: 0,
                        bottom: 0
                    },
                    coords: selectionInfo.rectArray
                })
                .then(annots => {
                    const annot = annots[0];
                    return this.getPDFUI().getPDFViewer().then(pdfViewer => {
                        pdfViewer.eventEmitter.emit(
                            PDFViewCtrl.ViewerEvents.addAnnotSuccess,
                            annot
                        );
                        pageRender.annotsRender.activeAnnot(annot.getId());
                    });
                });
        });
    }
}

# 修改组件

tooltip 涉及的组件名称：

名称 说明
fv--text-selection-tooltip 浮动框组件名称
fv--text-selection-tooltip-copy 复制选中文本内容
fv--text-selection-tooltip-create-highlight 复制选中文本内容
fv--text-selection-tooltip-create-strikeout 创建strikeout
fv--text-selection-tooltip-create-underline 创建underline

# 新增组件

向 tooltip 中插入创建 squiggly 的按钮

new UIExtension.PDFUI({
    //...
    fragments: [{
        target: 'fv--text-selection-tooltip',
        action: UIExtension.UIConsts.FRAGMENT_ACTION.APPEND,
        template: '<xbutton name="cus-tooltip-create-squiggly" icon-class="fv__icon-toolbar-squiggly"></xbutton>',
        config: {
            target: 'cus-tooltip-create-squiggly',
            callback: CreateSquigglyOnSelectedTextController
        }
    }]
})

# 删除组件

删除 underline 组件

new UIExtension.PDFUI({
    //...
    fragments: [{
        target: 'fv--text-selection-tooltip-create-underline',
        action: UIExtension.UIConsts.FRAGMENT_ACTION.REMOVE,
    }]
})

# 替换组件

将 underline 替换成创建 squiggly 的按钮

new UIExtension.PDFUI({
    //...
    fragments: [{
        target: 'fv--text-selection-tooltip-create-underline',
        action: UIExtension.UIConsts.FRAGMENT_ACTION.REPLACE,
        template: '<xbutton name="cus-tooltip-create-squiggly" icon-class="fv__icon-toolbar-squiggly"></xbutton>',
        config: {
            target: 'cus-tooltip-create-squiggly',
            callback: CreateSquigglyOnSelectedTextController
        }
    }]
})

# 重写浮动框组件模板

new UIExtension.PDFUI({
    //...
    fragments: [{
        target: 'fv--text-selection-tooltip',
        action: UIExtension.UIConsts.FRAGMENT_ACTION.REPLACE,
        template: `
            <tooltip-layer name="fv--text-selection-tooltip" class="fv--ui-text-selection-tooltip">
                <xbutton @controller="text-sel:CopySelectedTextController" icon-class="fv__icon-popup-copy"></xbutton>
                <xbutton @controller="text-sel:CreateTextHighlightOnSelectedTextController" icon-class="fv__icon-toolbar-text-highlight"></xbutton>
                <xbutton @controller="text-sel:CreateStrikeoutOnSelectedTextController" icon-class="fv__icon-toolbar-strikeout"></xbutton>
                <xbutton @controller="text-sel:CreateUnderlineOnSelectedTextController" icon-class="fv__icon-toolbar-underline"></xbutton>
                <xbutton name="cus-tooltip-create-squiggly" icon-class="fv__icon-toolbar-squiggly"></xbutton>
            </tooltip-layer>
        `,
        config: {
            target: 'cus-tooltip-create-squiggly',
            callback: CreateSquigglyOnSelectedTextController
        }
    }]
})
